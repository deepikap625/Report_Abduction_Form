<?php
$to = "palakondadeepika11@gamil.com"; // Replace with your email to receive the test
$subject = "Test Email from XAMPP using Gmail SMTP";
$message = "Hello! This is a test email sent from XAMPP on Windows.";

$headers .= "Reply-To: palakondadeepika11@gmail.com\r\n";

} else {
    echo " Failed to send mail.";
}
